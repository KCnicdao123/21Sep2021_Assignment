#include <iostream>
using namespace std;
int main()
{
	int grade = 120;
	if ((grade >= 70) && (grade <= 100)) {
		cout << "A" << endl;
	}
	else if ((grade >= 60) && (grade <= 69)) {
		cout << "B" << endl;
	}
	else if ((grade >= 50) && (grade <= 59)) {
		cout << "C" << endl;
	}
	else if ((grade >= 40) && (grade <= 49)) {
		cout << "D" << endl;
	}
	else if (grade < 40) {
		cout << "F" << endl;
	}
	else {
		cout << "Invalid input" << endl;
	}
}
